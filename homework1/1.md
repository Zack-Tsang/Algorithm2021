```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        for i in range(len(nums)):
            pre = nums[i]
            res = nums[i+1:]
            if (target - pre) in res:
                return [i, res.index(target - pre) + i + 1]
```
